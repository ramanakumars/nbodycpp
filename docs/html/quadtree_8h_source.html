195 void calculateCOM() {
196 centerOfMass = {0, 0};
197 totalMass = 0;
198
199 if (is_divided) {
200 for (const auto &child : children) {
201 child->calculateCOM();
202 totalMass += child->totalMass;
203 centerOfMass += (child->centerOfMass * child->totalMass);
204 }
205 centerOfMass /= totalMass;
206 } else {
207 for (auto &particle : particles) {
208 centerOfMass = (centerOfMass * totalMass + particle->position * particle->mass) /
209 (totalMass + particle->mass);
210 totalMass += particle->mass;
211 }
212 }
